function encontrarMaiorNumero(string) { const numeros = string.match(/\d+/g); if (!numeros || numeros.length === 0) { return NaN; } return Math.max(...numeros.map(Number)); } const string = "Este é um exemplo 123 com o número 987"; const maior = encontrarMaiorNumero(string); console.log(`O maior número na string é: ${maior}`);
